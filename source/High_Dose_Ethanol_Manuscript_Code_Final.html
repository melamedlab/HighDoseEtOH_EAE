comparison.plot <- function(pattern1, pattern2, figure_name) {
  
  stopifnot(pattern1 %in% names(group_colors),
            pattern2 %in% names(group_colors))
  
  col1 <- group_colors[pattern1]
  col2 <- group_colors[pattern2]
  
  gr1 <- eae_data %>%
    filter(str_detect(pattern, pattern1)) %>%
    select(`7`:`19`)
  
  gr2 <- eae_data %>%
    filter(str_detect(pattern, pattern2)) %>%
    select(`7`:`19`)
  
  gr1.means <- summarise(gr1, across(everything(), mean, na.rm = TRUE)) %>% unlist()
  gr1.stdev <- summarise(gr1, across(everything(), standard_error)) %>% unlist()
  
  gr2.means <- summarise(gr2, across(everything(), mean, na.rm = TRUE)) %>% unlist()
  gr2.stdev <- summarise(gr2, across(everything(), standard_error)) %>% unlist()
  
  sigs <- mapply(t_test_p_value, gr1, gr2)
  days <- as.numeric(colnames(gr1))
  
  pdf(paste(output, figure_name, ".pdf"),
      width = 6, height = 4)
  
  plot(days, gr1.means, type = "o",
       col = col1, frame.plot = FALSE, axes = FALSE,
       xlim = c(7, 19), ylim = c(0, 4),
       xlab = "Day", ylab = "Score")
  
  axis(2, at = 0:4)
  axis(1, at = 7:19)
  
  lines(days, gr2.means, type = "o", col = col2)
  
  errbar(days, gr1.means,
         gr1.means + gr1.stdev,
         gr1.means - gr1.stdev,
         add = TRUE, errbar.col = col1, col = col1)
  
  errbar(days, gr2.means,
         gr2.means + gr2.stdev,
         gr2.means + (-gr2.stdev),
         add = TRUE, errbar.col = col2, col = col2)
  
  sel <- (0.01 < sigs & sigs < 0.05)
  text(days[sel], pmax(gr1.means[sel], gr2.means[sel]) + 0.45, "*", cex = 1.5)
  
  sel <- (0.001 < sigs & sigs < 0.01)
  text(days[sel], pmax(gr1.means[sel], gr2.means[sel]) + 0.45, "**", cex = 1.5)
  
  sel <- sigs < 0.001
  text(days[sel], pmax(gr1.means[sel], gr2.means[sel]) + 0.45, "***", cex = 1.5)
  
  legend("topleft",
         legend = c(pattern1, pattern2),
         col = c(col1, col2),
         lty = 1,
         pch = 16,
         bty = "n")
  
  dev.off()
}

read_profile <- function(file) {
  read_tsv(file, comment = "#",
           col_names = c("clade", "tax_id", "abundance", "extra"),
           col_types = cols(.default = "c")) %>%
    mutate(
      abundance = as.numeric(abundance),
      sample_id = basename(file) %>%
        str_remove("_taxonomic_profile\\.tsv") %>%
        str_replace("pre-EAE", "-D0") %>%
        str_replace("post-EAE", "-D26")
    ) %>%
    filter(!is.na(abundance)) %>%
    select(taxon = clade, abundance, sample_id)
}
